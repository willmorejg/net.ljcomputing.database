017package net.ljcomputing.database.servcie.impl;
018
019import freemarker.ext.beans.BeansWrapper;
020import freemarker.ext.beans.BeansWrapperBuilder;
021import freemarker.template.Configuration;
022import freemarker.template.TemplateHashModel;
023
024import java.io.BufferedOutputStream;
025import java.io.File;
026import java.io.FileOutputStream;
027import java.io.OutputStream;
028import java.util.HashMap;
029import java.util.Map;
030
031/**
032 * Abstract implementation of a database conversion service.
033 * 
034 * @author James G. Willmore
035 *
036 */
037public abstract class AbstractDatabaseConversionService {
038
039  /** The output directory path. */
040  protected String outputDirectoryPath;
041
042  /** The output directory. */
043  protected File outputDirectory;
044
045  /** The Freemarker configuration. */
046  protected Configuration freemarkerConfiguration;
047
048  /** The root. */
049  protected Map<String, Object> root;
050
051  /** The wrapper. */
052  protected BeansWrapper wrapper;
053
054  /** The static models. */
055  protected TemplateHashModel staticModels;
056
057  /** The string utils. */
058  protected TemplateHashModel stringUtils;
059
060  /**
061   * Instantiates a new abstract database conversion service.
062   *
063   * @param outputDirectoryPath the output directory path
064   * @param freemarkerConfiguration the freemarker configuration
065   * @throws Exception the exception
066   */
067  AbstractDatabaseConversionService(String outputDirectoryPath, 
068      Configuration freemarkerConfiguration)
069      throws Exception {
070    this.outputDirectoryPath = outputDirectoryPath + "/";
071    this.freemarkerConfiguration = freemarkerConfiguration;
072    this.outputDirectory = new File(this.outputDirectoryPath);
073
074    root = new HashMap<String, Object>();
075    wrapper = new BeansWrapperBuilder(Configuration.VERSION_2_3_21).build();
076
077    staticModels = wrapper.getStaticModels();
078    stringUtils = (TemplateHashModel) staticModels.get("net.ljcomputing.StringUtils");
079
080    root.put("stringUtils", stringUtils);
081  }
082
083  /**
084   * Write to file.
085   *
086   * @param file the file @param content the content
087   * @param content the content
088   * @throws Exception the exception
089   */
090  protected void writeToFile(File file, String content) throws Exception {
091    OutputStream os = new BufferedOutputStream(new FileOutputStream(file));
092    os.write(content.getBytes());
093    os.flush();
094    os.close();
095  }
096}

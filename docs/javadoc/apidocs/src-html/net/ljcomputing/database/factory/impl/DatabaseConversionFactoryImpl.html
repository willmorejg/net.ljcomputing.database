017package net.ljcomputing.database.factory.impl;
018
019import freemarker.template.Configuration;
020
021import net.ljcomputing.database.context.ConversionServiceContext;
022import net.ljcomputing.database.context.impl.DatabaseConversionServiceContext;
023import net.ljcomputing.database.factory.DatabaseConversionFactory;
024import net.ljcomputing.database.servcie.DatabaseConversionService;
025import net.ljcomputing.database.servcie.impl.DatabaseUsingTemplateConverter;
026import net.ljcomputing.database.strategy.DatabaseConversionStrategy;
027import net.ljcomputing.database.strategy.DatabaseConversionStrategyType;
028
029import org.slf4j.Logger;
030import org.slf4j.LoggerFactory;
031
032import org.springframework.beans.factory.annotation.Autowired;
033import org.springframework.beans.factory.annotation.Value;
034import org.springframework.stereotype.Service;
035
036/**
037 * Database conversion service factory - creates a conversion service based upon
038 * the strategy type requested.
039 *
040 * @author James G. Willmore
041 */
042@Service
043public class DatabaseConversionFactoryImpl
044    implements DatabaseConversionFactory {
045
046  /** SLF4J The logger. */
047  private static Logger logger = LoggerFactory
048      .getLogger(DatabaseConversionFactoryImpl.class);
049
050  /** The instance. */
051  private static DatabaseConversionFactoryImpl instance;
052
053  /** The output path. */
054  @Value("${database.output.path:out}")
055  private String outputDirectoryPath;
056
057  /** The Freemarker configuration. */
058  @Autowired
059  private Configuration freemarkerConfiguration;
060
061  /**
062   * Instantiates a new database conversion factory impl.
063   */
064  private DatabaseConversionFactoryImpl() {
065    super();
066  }
067
068  /**
069   * Gets the single instance of the database conversion service factory.
070   *
071   * @return single instance of the database conversion service factory.
072   */
073  public static synchronized DatabaseConversionFactoryImpl getInstance() {
074    if (null == instance) {
075      instance = new DatabaseConversionFactoryImpl();
076    }
077
078    return instance;
079  }
080
081  /**
082   * Creates the conversion service.
083   *
084   * @param conversionStrategy the conversion strategy
085   * @return the database conversion service
086   * @throws Exception the exception
087   * @see net.ljcomputing.database.factory.DatabaseConversionFactory
088   * #createConversionService(net.ljcomputing.database.strategy.DatabaseConversionStrategyType.Type)
089   */
090  public DatabaseConversionService createConversionService(
091      DatabaseConversionStrategyType.Type conversionStrategy) throws Exception {
092    DatabaseConversionStrategy strategy = conversionStrategy.strategy();
093    ConversionServiceContext ctx = strategy.getContext();
094    ConversionServiceContext context = new DatabaseConversionServiceContext(
095        outputDirectoryPath, freemarkerConfiguration, ctx.getOutputTemplate(),
096        ctx.getFileSuffix());
097
098    logger.debug("context : {}", context);
099    return new DatabaseUsingTemplateConverter(context);
100  }
101}
